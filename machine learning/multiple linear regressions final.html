\( E = y^T y - 2y^T X\beta + \beta^T X^T X \beta \)
\( \frac{\partial E}{\partial \beta} = \frac{d}{d\beta} \left( y^T y - 2y^T X\beta + \beta^T X^T X \beta \right) \)
\( \Rightarrow 0 - 2y^T X + \frac{d}{d\beta} \left[ \beta^T X^T X \beta \right] = 0 \)
\( \Rightarrow -2 y^T X + 2 \beta^T X^T X \)
\( \frac{\partial E}{\partial \beta} = \beta^T X^T X = y^T X \)
\( \Rightarrow (\beta^T X^T X) (X^T X)^{-1} = y^T X (X^T X)^{-1} \)
\( \Rightarrow \beta^T I = y^T X (X^T X)^{-1} \)
\( \Rightarrow \beta^T = y^T X (X^T X)^{-1} \)
\( (\beta^T)^T = \left[ y^T X (X^T X)^{-1} \right]^T \)
\( \Rightarrow \beta = \left[ (X^T X)^{-1} X^T y \right]^T \)
\( \Rightarrow \beta = (X^T X)^{-1} X^T y \)
\(\displaystyle y^T \hat{y} = \hat{y}^T y \quad \text{(proof)}\)
Proof: Let \( y = A, \quad \hat{y} = B \)
\( \Rightarrow (A^T B)^T = (B^T A) \)
\( \Rightarrow (A^T B)^T = A^T B \)
\( \Rightarrow A^T B = C \)
\( \Rightarrow A^T B \text{ is symmetric} \)
\( \Rightarrow y^T \hat{y} = y^T X \)
